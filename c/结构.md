# 结构

### struct

```c
// point是结构标记
// 这只是结构声明并没有分配内存空间
struct point {
    int x;
    int y;
};

// 分配内存空间
struct point {
    int x;
    int y;
} p;

struct point p; // 声明
// struct point p在计算机看来等同于下面
struct point { 
    int x;
    int y;
} p;

// 多次使用就带上结构标记
struct {
    int x;
    int y;
} p;

// x, y, z是结构变量表
// 和声明int x, y, z是一样的
// 并且为它们分配存储空间
struct point {
    int x;
    int y;
} x, y, z;

// 结构数组
struct point {
    int x;
    int y;
} x[100];

// 初始化
struct point x = {.x = 1, .y = 2};

// 伸缩型数组成员 C99
// 该数组不会立即存在（sizeof()时不会计算空间）
// 伸缩型数组成员必须是最后一个
// C99的意图不让让你声明struct p类型的变量，而是希望你声明一个指向struct p类型的指针
// malloc(sizeof(struct p) + 5 * sizeof(int));
struct p {
    // 声明时，不能做任何事情，因为没有预留存储空间
    int arr[]; 
};
```

### union

```c
// 在不同时刻保存不同的类型和长度的对象变量
// 存储结构是最宽的属性
union u_tag {
    int ival;

    char *sval;
```

### 位字段

```c
// 4 bytes
struct {
    unsigned int a : 1;
    unsigned int b : 1;
    unsigned int c : 1;
} flags;

// 不允许一个跨越两个unsigned
// 可以使用未命名的字段宽度填充
// 使用一个宽度位0的未命名字段迫使下一个字段与下一个整数对齐
// 8 bytes
struct {
    unsigned int a : 1;
    unsigned int   : 0;
    unsigned int b : 1;
}
```

包含3个一位的字段，冒号后面表示字段的宽度（用二进制位数表示）
