# 模版与泛型

### 函数模版

```cpp
// typename 也可以写成class
// 也可以用typename用作函数模版，class写作类模版
template <typename T>
int compare(const T &v1, const T &v2) {}

// 实例化类模板
compare<int>(1, 2);

// 具体化模版，比如对比两个S类
template<> int compoare(S s1, S s2) {}

// 强制调用模版函数
compare<>(1, 2);
```

### 类模版

```cpp
template <class NameType,class AgeType> 
class S 
{
  public:
  NameType name_;
  AgeType age_;
  
  S(NameType name, AgeType age)
  {
    this->name = name;
    this->age = age;
  }
    
  void p();
}

template <class NameType,class AgeType> 
void S<NameType, AgeType>::p()
{
    
}

// 类模版没有类型自动推导，必须显示指定
S<string, int> s1("12", 10);
```
