# 容器

### vector

```cpp
// 可变大小数组，支持快速随机访问
// 在尾部之外的地方插入和删除很慢
// vector对象被销毁时，存储在其中的对象也会被销毁
#include <vector>
using std::vector;

// 10个元素，默认值初始化，如果是int类型为0，string为空
vector<int> ivec(10);

vector<int> v;

for (auto value : v) 
  cout << value << endl;

for (vector<int>::itertor it = v.begin(); it != v.end(); it++)
  cout << *it << endl;

// 尾插 尾删
v.push_back(1);
v.pop_back();

v.insert(v.begin(), 100);
v.erase(v.begin);
v.clear();

v.at(1);
v[1];

// 返回第一个/最后一个元素
v.front();
v.back();

// 可以缩容
v.sawp(v1);

v.capacity();
v.size();

// 只影响size不影响capacity
v.resize(1);

// 预留空间
v.reserve(10);
```

### deque

```cpp
// 双端队列，支持快速随机访问
// 在头尾位置插入/删除速度快
// 没有容量的概念
#include <deque>
using std::deque;

deque<int> q;

q.push_back(1);
q.pop_back();

q.push_front(1);
q.pop_front();

q.insert(q.begin(), 1);
```

### list

```cpp
// 双向链表，只支持双向顺序访问
// 在任何位置插入/删除都很快
#include <list>
using std::list;

list<int> l;

// 单向链表，只支持单向顺序访问
// 在任何位置插入/删除速度快
#include <forward_list>
using std::forward_list;

forward_list <int> l;
```

### array

// 固定大小数组，支持快速随机访问

### map

```cpp
// 关联数组
#include <map>
using std::map;

map<string, int> m = {
  {"key", 1},
  {"key1", 2}
};

m.insert({"123", 1});
m.count(key);
m.find(key);

// 关键字可重复出现的map
#include <multimap>
using std::multimap;

multimap<string, int> m;

// 哈希函数组织的map
#include <unordered_map>
unordered_map<string, int> m;

// 哈希函数组织的map，key可以重复
#include <unordered_map>
unordered_multimap<string, int> m;
```

### set

```cpp
#include <set>
using std::set;

// set 只保存关键字的容器
set<string> s = {"test", "test1"};
s.cout(key);
s.find(key);

// 关键字可重复
multiset<string> s;

// 只有insert
s.insert("123");

// 哈希函数组织的set
#include  <unordered_set>
unordered_set <string> s;

// 哈希函数组织的set，关键子可以重复
#include <unordered_set>
unordered_multiset<string> s;
```
